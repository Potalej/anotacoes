POSICOES, MOMENTOS = [R0], [P0]
R, P = R0, P0

for _ in range(QNTD):
    
    # Calcula as forcas
    Fsomas = np.zeros((N,3))
    for a in range(N):
        ma = m[a]
        ra = R[a]
        for b in range(N):
            if a == b: continue
            mb = m[b]
            rb = R[b]
            rab = np.linalg.norm(rb - ra)
            Fab = mb * (rb - ra)/(rab**3)
            
            Fsomas[a] += Fab

    R = R + h * P * m_inv
    P = P + h * Fsomas

    POSICOES.append(R)
    MOMENTOS.append(P)

Um problema aparente é a integração em função de ζζ, pois uma vez que o sistema original fornece um DD que não tem variação constante o tamanho do passo deveria se alterar em cada integração para garantir uma correspondência exata. Isso pode ser ignorado tomando a perspectiva não do sistema original, mas do sistema normalizado, e a correspondência das órbitas podem ser vistas simplesmente pelo formato das órbitas, pois trata-se de um caso simples.

SIGMAS, PIS = [SIGMAS0], [PIS0]
SIGMA, PI = SIGMAS0, PIS0

for t in range(10*QNTD):
    # Acrescimo manual e constante
    zeta = t*h

    # Calcula as forcas
    M = np.sum(m)
    Fsomas = np.zeros((N,3))
    for a in range(N):
        ma = m[a]
        mua = ma/M
        ra = SIGMA[a]
        for b in range(N):
            if a == b: continue
            mb = m[b]
            mub = mb/M
            rb = SIGMA[b]
            
            vetor_dist = rb * (1/np.sqrt(mub)) - ra * (1/np.sqrt(mua))
            rab = np.linalg.norm(vetor_dist)
            
            vetor_dist = ra / mua - rb / np.sqrt(mub*mua)
            Fab = ma * mb * vetor_dist/(rab**3)
            
            Fsomas[a] += Fab

    ks = auxiliares.KS(PI)
    vs = auxiliares.VS(m, SIGMA)

    SIGMA_1 = SIGMA + h * 2 * PI / (ks + zeta**2)
    PI_1    = PI + h * Fsomas / vs - h * ks/(ks + zeta**2) * SIGMA

    SIGMA = SIGMA_1
    PI = PI_1

    SIGMAS.append(SIGMA)
    PIS.append(PI)
